#include <iostream>

using namespace std;

int main()
{
	int n = 0;

	cin >> n;

	if (n > 0)
	{
		// Выделяем матрицу 2D
		int** ppArr = new int * [n]; // 1

		for (size_t i = 0; i < n; i++) // 1
			ppArr[i] = new int[n];

		// 2
		ppArr[0][0] = 10.0f;
		ppArr[1][1] = ppArr[0][0] + 10.0f;

		for (size_t i = 0; i < n; i++) // 3
			delete [] ppArr[i];

		delete[] ppArr; // 3
	}

	return 0;
}
