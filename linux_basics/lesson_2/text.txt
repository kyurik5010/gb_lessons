Ctrl+Shift+t               - новая вкладка терминала

sudo systemctl stop sshd   - команда остановки SSH
sudo systemctl status sshd - команда проверки статуса SSH
sudo systemctl start sshd  - команда запуска SSH

iTerm2 - SSH-клиент для MacOS

tree -d         - выводит древо директорий и поддиректорий
tree -d -L l    - выводит древо директорий 1го уровня 
tree -d -L 1 /  - выводит древо директорий корневого каталога
tree -d -L 1 .  - выводит древо директорий текущего каталога

pwd             - print working directory
cd -            - перемещение в предыдущий каталог (хоть в другой ветке)
cd / cd ~       - перемещение в корневой каталог

полный путь - это путь от корневого каталога
относительный путь - неполный путь


Ctrl+l    - очистка терминала - аналог clear
Ctrl+r    - посик по истории команд (варианты перебираются при повторном 
            нажатии комбинации)
Ctrl+w    - удаление слова слева от курсора
Ctrl+u    - удаление всего слева от курсора
Ctrl+k    - удалнеие всего справа от курсора

w (who)   - посмотреть кто есть в системе
id        - имя пользователя (whoami)

ip a      - посмотреть IP адрес
ping      - пропинговка адреса (останавливается Ctrl+C)
sleep 5;  - задержка на 5 сек. перед выполнением последующей команды 

sudo -i   - переход в рут

ls -l     - больше информации о файлах каталога
ls -lh    - то же самое но human readable 
ls -a     - показывает скрытые файлы

cp        - копирование файлов
cp -v     - копирование с отчетом (verbose)
cp -r     - рекурсивное копирование (для каталогов)

mv        - переименование/перемещение, ключи те же что у 'cp', кроме '-r', т.к.
            операция 'mv' никак не воздействует на каталоги

rm        - удаление, самая опасная команда (можно систему так снести)
rm -f     - удаление с ключом 'forced' насильно

mkdir -p  - 'parent' создание ролительских каталогов, например:
            'mkdir -p dir1/dir2 dir1/dir3' 

cat       - вывод содержимого файла
more      - вывод содержимого файла постранично
less      - вывод содержимого файла построчно
tail      - вывод последних 10 строк файла
tail -f   - вывод посл. 10 строк с ключом 'follow', которая 
            отслеживает изменения в файле (например файл лога)
            пример: 'tail -f /var/log/syslog'
head      - вывод первые 10 строк файла

|         - конвейер, пример: 'cat dir1/text.txt | less',
            т.е. команда по конвейеру передаётся в less

-r  - рекурсивное действие (в основном для каталогов используется)
-y  - везде отвечать yes / удобно использовать с sudo apt install <Имя> -y
-v  - verbose, режим с фидбеком
-i  - ключ интереактивности (do you want to delete this file? (y/n)
