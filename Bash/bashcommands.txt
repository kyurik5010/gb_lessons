ls -a (показать скрытые файлы папки)
cd ~ (возвращение в корневую директорию)
cd .. (возвращение в предыдущую директорию)
cd ../.. (возвращение на 2 директории назад)
mkdir имя (создание директории) 
touch имя (создание файла, нужно прописывать расширение)
echo "..." > имя (запись строки в файл)
cat имя (чтение из файла)
rm имя (удаление файла, минуя корзину)
rm -R имя (удаление директории с подтверждением) 
rmdir имя (удаление директории)
mv имя имя (переименовывает файлы)
